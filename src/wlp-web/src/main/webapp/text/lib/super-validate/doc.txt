//绑定一个表单的验证事件
validateInput('entitystudentname', function(id, val, obj) {
// 学生姓名
	if (valid_isNull(val)) {
		return {
			valid : false,
			msg : '不能为空'
		};
	}
	if (valid_maxLength(val, 12)) {
		return {
			valid : false,
			msg : '长度不能大于' + 12
		};
	}
	return {
		valid : true,
		msg : '正确'
	};
}, 'errorMsgBoxId', 'bootstrapFormGroupId');
